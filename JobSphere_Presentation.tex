% ===== SECTION 6: JOB APPROVAL WORKFLOW =====
\section{Job Approval Workflow}

\begin{frame}
    \frametitle{Workflow: User Posts Job}
    \centering
    \begin{enumerate}
        \item User fills job form: title, company, category, description, apply link
        \item Clicks \textbf{``Publish''} button
        \item System creates job with \textcolor{orange}{\textbf{status = 'pending'}}
        \item Job saved to storage
        \item Admin receives notification
        \item Admin reviews job in \textbf{Pending} tab
        \item Admin clicks \textbf{``Approve''} or \textbf{``Reject''}
        \item If approved: status becomes \textcolor{JobSphereGreen}{\textbf{approved}}
        \item Job appears in public listings
    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{Workflow: Admin Posts Job}
    \centering
    \begin{enumerate}
        \item Admin logs in to \textbf{Admin Dashboard}
        \item Goes to \textbf{``Post Job''} tab
        \item Fills job details: title, company, category, location, description, link
        \item Clicks \textbf{``Post Job (Auto-Approved)''}
        \item System creates job with \textcolor{JobSphereGreen}{\textbf{status = 'approved'}}
        \item Job IMMEDIATELY appears in public listings
        \item NO waiting period
        \item Users can apply instantly
    \end{enumerate}
\end{frame}

% ===== SECTION 10: PRODUCTION DEPLOYMENT =====
\section{Production Deployment}

\begin{frame}
    \frametitle{From Development to Production}
    \begin{center}
        \textbf{Migration Path: localStorage → MongoDB}
    \end{center}
    \vspace{0.3cm}
    \begin{enumerate}
        \item \textbf{Step 1:} Set up MongoDB cluster on MongoDB Atlas
        \item \textbf{Step 2:} Deploy backend (Node.js + Express) to Heroku/Vercel
        \item \textbf{Step 3:} Configure environment variables
        \begin{itemize}
            \item \texttt{MONGODB\_URI}
            \item \texttt{JWT\_SECRET}
        \end{itemize}
        \item \textbf{Step 4:} Enable MongoDB in frontend: set \texttt{DB\_CONFIG.USE\_MONGODB = true}
        \item \textbf{Step 5:} Test API endpoints
        \item \textbf{Step 6:} Deploy frontend to Netlify
    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{Security Considerations}
    \begin{center}
        \textbf{Before Production Deployment}
    \end{center}
    \vspace{0.3cm}
    \begin{itemize}
        \item<1-> \textbf{Password hashing:} Implement bcrypt (backend ready)
        \item<2-> \textbf{JWT tokens:} Auto-generated with 7-day expiry
        \item<3-> \textbf{CORS:} Configured for production domains
        \item<4-> \textbf{Rate limiting:} Prevent spam submissions
        \item<5-> \textbf{Email verification:} Optional for user accounts
        \item<6-> \textbf{HTTPS:} Required for production
        \item<7-> \textbf{Environment variables:} Never hardcode secrets
        \item<8-> \textbf{Input validation:} Server-side checks
    \end{itemize}
\end{frame}
